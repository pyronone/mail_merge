note on tables

For tables, the original data would typically look something like this:

data = {
    "id": [99, 99, 100, 100, 100, 100],
    "type": ["A", "C", "A", "B", "A", "B"],
    "amt": [10, 20, 30, 40, 50, 60],
}
df = pd.DataFrame(data)

# * DOC
def _create_random_benes(id: str) -> pd.DataFrame:
    from faker import Faker

    fake = Faker()
    df = pd.DataFrame(columns=["id", "name", "pct"])

    num_benes = random.randint(1, 4)

    for i in range(num_benes):
        df.loc[len(df)] = [id, fake.name(), f"{round(100/num_benes,2)}%"]

    return df


if REFRESH:
    df = pd.read_parquet("../tests2/clean_dat.parquet")
    dfs = []
    for _id in df["id"].tolist():
        dfs.append(_create_random_benes(_id))
    benes = pd.concat(dfs)

    benes["num"] = benes.groupby("id").cumcount() + 1
    benes_pivot = benes.pivot(index="id", columns="num", values=["name", "pct"])
    benes_pivot.columns = [f"__tbl2__{col[0]}_{col[1]}" for col in benes_pivot.columns]
    benes_pivot = benes_pivot.reset_index().copy()

    tm = pd.merge(df, benes_pivot, "inner", "id")
    assert len(tm) == len(df)
    assert len(tm) == len(benes_pivot)

    tm.to_parquet("../tests2/clean_dat2.parquet")

Config is in yaml format:

The config file is used to specify the data source (currently, only parquet is supported to ensure consistent/persistent data types), groups, and what template(s) each group should use.
